### Paso 2: Código de ejemplo en Python

Este código hará preguntas a ChatGPT sobre Coca-Cola usando la API:

```python
import os
import openai

# Colocar tu API Key aquí o usar variables de entorno
openai.api_key = os.getenv("OPENAI_API_KEY")  # recomendado, sino poner la clave directamente

def preguntar_cocacola(pregunta):
    prompt = f"Eres un asistente experto en la empresa Coca-Cola. Responde a la siguiente pregunta de forma clara y concisa:\n\n{pregunta}"

    response = openai.ChatCompletion.create(
        model="gpt-4",  # o "gpt-3.5-turbo"
        messages=[
            {"role": "system", "content": "Eres un asistente experto en la empresa Coca-Cola."},
            {"role": "user", "content": pregunta}
        ],
        max_tokens=500,
        n=1,
        stop=None,
        temperature=0.7,
    )

    respuesta = response.choices[0].message['content'].strip()
    return respuesta

if __name__ == "__main__":
    print("Chatbot Coca-Cola (escribe 'salir' para terminar)")
    while True:
        pregunta = input("Tu pregunta: ")
        if pregunta.lower() == "salir":
            break
        respuesta = preguntar_cocacola(pregunta)
        print("Bot:", respuesta)
```

### Archivo `app.py`

```python
import os
from flask import Flask, request, render_template_string
import openai
from dotenv import load_dotenv

# Carga variables de entorno
load_dotenv()

openai.api_key = os.getenv("OPENAI_API_KEY")

app = Flask(__name__)

HTML_TEMPLATE = """
<!doctype html>
<title>Chatbot Coca-Cola</title>
<h1>Chatbot Coca-Cola - Pregúntame lo que quieras</h1>
<form method=post>
  <textarea name=question rows=4 cols=50 placeholder="Escribe tu pregunta aquí..."></textarea><br>
  <input type=submit value=Enviar>
</form>
{% if answer %}
<h2>Respuesta:</h2>
<p>{{ answer }}</p>
{% endif %}
"""

def obtener_respuesta(pregunta):
    messages = [
        {"role": "system", "content": "Eres un experto en la empresa Coca-Cola, responde brevemente."},
        {"role": "user", "content": pregunta}
    ]

    respuesta = openai.ChatCompletion.create(
        model="gpt-3.5-turbo",
        messages=messages,
        max_tokens=500,
        temperature=0.7,
    )
    return respuesta.choices[0].message['content'].strip()

@app.route("/", methods=["GET", "POST"])
def index():
    answer = None
    if request.method == "POST":
        pregunta = request.form.get("question")
        if pregunta:
            answer = obtener_respuesta(pregunta)
    return render_template_string(HTML_TEMPLATE, answer=answer)

if __name__ == "__main__":
    app.run(debug=True)
```
